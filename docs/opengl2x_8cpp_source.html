638  if( dev->isTileRenderStarted )
639  return;
640 
641  if( dev->hasDiscardBuffers && (dev->target.color[1]==0) &&
642  dev->target.color[0] ){
643  GLenum flg[3] = {}, *p = flg;
644 
645  if( !dev->target.color[0] || !dev->target.color[0]->isInitalized ){
646  *p = GL_COLOR_ATTACHMENT0; ++p;
647  }
648  if( !dev->target.depth || !dev->target.depth->isInitalized ){
649  *p = GL_DEPTH_ATTACHMENT; ++p;
650  }
651 
652  if(p-flg>0)
653  dev->glDiscardFrameBuffer(GL_FRAMEBUFFER, p-flg, flg);
654  errCk();
655  } else
656  if( dev->hasQCOMTiles ){
657  int w = dev->scrW, h = dev->scrH;
658  (void)w;
659  (void)h;
660 
661  GLbitfield flg = 0;
662  GLbitfield nflg = GL_COLOR_BUFFER_BIT0_QCOM;
663 
664  for( int i=0; i<dev->caps.maxRTCount; ++i )
665  if( dev->target.color[i] ){
666  w = dev->target.color[i]->w;
667  h = dev->target.color[i]->h;
668  nflg = GL_NONE;
669 
670  if( dev->target.color[i]->isInitalized )
671  flg |= GL_COLOR_BUFFER_BIT0_QCOM;
672  }
673 
674  flg|=nflg;
675 
676  if( (dev->target.depth && dev->target.depth->isInitalized)||
677  !dev->target.depth )
678  flg |= GL_DEPTH_BUFFER_BIT0_QCOM;
679
